 87class OIDCJWTVerifier(TokenVerifier):
 88    """JWT verifier for resource-server deployments using an external OIDC issuer."""
 89
 90    def __init__(self, config: RemoteAuthConfig) -> None:
 91        try:
 92            import jwt
 93            from jwt import PyJWKClient
 94        except ImportError as exc:  # pragma: no cover - exercised only without remote extra
 95            raise RuntimeError(
 96                "Install anthropic-news-mcp[remote] to run Streamable HTTP with OIDC/JWT auth."
 97            ) from exc
 98
 99        self._jwt: Any = jwt
100        self._jwks = PyJWKClient(f"{config.issuer.rstrip('/')}/.well-known/jwks.json")
101        self._config = config
102
103    async def verify_token(self, token: str) -> AccessToken | None:
104        loop = asyncio.get_event_loop()
105        try:
106            signing_key = await loop.run_in_executor(
107                None, self._jwks.get_signing_key_from_jwt, token
108            )
109        except Exception:
110            _log.warning(
111                "JWKS key lookup failed — all tokens will be rejected until resolved",
112                exc_info=True,
113            )
114            return None
115        try:
116            claims = self._jwt.decode(
117                token,
118                signing_key.key,
119                algorithms=["RS256", "ES256"],
120                audience=self._config.audience,
121                issuer=self._config.issuer,
122                options={"require": ["exp", "iat"]},
123            )
124        except Exception:
125            return None
126
127        scopes = _extract_scopes(claims)
128        expires_at = claims.get("exp")
129        client_id = str(claims.get("client_id") or claims.get("azp") or claims.get("sub") or "")
130        if not client_id:
131            return None
132        return AccessToken(
133            token=token,
134            client_id=client_id,
135            scopes=scopes,
136            expires_at=expires_at if isinstance(expires_at, int) else None,
137            resource=self._config.resource_server_url,
138        )

141def _extract_scopes(claims: dict[str, Any]) -> list[str]:
142    raw_scope = claims.get("scope", "")
143    scopes: set[str] = set(raw_scope.split()) if isinstance(raw_scope, str) else set()
144    raw_scopes = claims.get("scp", [])
145    if isinstance(raw_scopes, str):
146        scopes.update(raw_scopes.split())
147    elif isinstance(raw_scopes, list):
148        scopes.update(str(scope) for scope in raw_scopes)
149    return sorted(scopes)

180class _TokenBucket:
181    def __init__(self, *, rate_per_minute: int, burst: int) -> None:
182        self.rate_per_second = rate_per_minute / 60
183        self.burst = float(burst)
184        self.tokens: dict[str, tuple[float, float]] = {}
185
186    def allow(self, key: str) -> bool:
187        now = time.monotonic()
188        tokens, updated_at = self.tokens.get(key, (self.burst, now))
189        tokens = min(self.burst, tokens + (now - updated_at) * self.rate_per_second)
190        if tokens < 1:
191            self.tokens[key] = (tokens, now)
192            return False
193        self.tokens[key] = (tokens - 1, now)
194        return True

197class RequestLogRateLimitMiddleware(BaseHTTPMiddleware):
198    def __init__(
199        self,
200        app: Any,
201        *,
202        rate_limit_per_minute: int,
203        rate_limit_burst: int,
204    ) -> None:
205        super().__init__(app)
206        self.bucket = _TokenBucket(
207            rate_per_minute=rate_limit_per_minute,
208            burst=rate_limit_burst,
209        )
210
211    async def dispatch(self, request: Request, call_next: Any) -> Response:
212        request_id = request.headers.get("x-request-id") or uuid4().hex
213        client = request.client.host if request.client else "unknown"
214        if not self.bucket.allow(client):
215            _log.info(
216                "remote_request",
217                extra={
218                    "request_id": request_id,
219                    "method": request.method,
220                    "path": request.url.path,
221                    "status_code": 429,
222                    "client": client,
223                },
224            )
225            return JSONResponse(
226                {"error": {"code": "rate_limited", "message": "Too many requests.", "details": {}}},
227                status_code=429,
228                headers={"x-request-id": request_id},
229            )
230        started = time.perf_counter()
231        response = cast(Response, await call_next(request))
232        response.headers["x-request-id"] = request_id
233        _log.info(
234            "remote_request",
235            extra={
236                "request_id": request_id,
237                "method": request.method,
238                "path": request.url.path,
239                "status_code": response.status_code,
240                "duration_ms": int((time.perf_counter() - started) * 1000),
241                "client": client,
242            },
243        )
244        return response

247class HostOriginMiddleware(BaseHTTPMiddleware):
248    def __init__(
249        self,
250        app: Any,
251        *,
252        allowed_hosts: set[str],
253        allowed_origins: set[str],
254    ) -> None:
255        super().__init__(app)
256        self.allowed_hosts = allowed_hosts
257        self.allowed_origins = allowed_origins
258
259    async def dispatch(self, request: Request, call_next: Any) -> Response:
260        host = request.headers.get("host", "").split(":", 1)[0]
261        origin = request.headers.get("origin")
262        request_id = request.headers.get("x-request-id") or uuid4().hex
263        if host and host not in self.allowed_hosts:
264            _log.warning(
265                "remote_request_denied",
266                extra={
267                    "request_id": request_id,
268                    "reason": "host_not_allowed",
269                    "host": host,
270                    "origin": origin,
271                    "path": request.url.path,
272                },
273            )
274            return JSONResponse(
275                {
276                    "error": {
277                        "code": "host_not_allowed",
278                        "message": "Host is not allowed.",
279                        "details": {},
280                    }
281                },
282                status_code=403,
283                headers={"x-request-id": request_id},
284            )
285        if origin and origin not in self.allowed_origins:
286            _log.warning(
287                "remote_request_denied",
288                extra={
289                    "request_id": request_id,
290                    "reason": "origin_not_allowed",
291                    "host": host,
292                    "origin": origin,
293                    "path": request.url.path,
294                },
295            )
296            return JSONResponse(
297                {
298                    "error": {
299                        "code": "origin_not_allowed",
300                        "message": "Origin is not allowed.",
301                        "details": {},
302                    }
303                },
304                status_code=403,
305                headers={"x-request-id": request_id},
306            )
307        response = await call_next(request)
308        return cast(Response, response)
