 61class NewsItem(BaseModel):
 62    """A single news/changelog/release item."""
 63
 64    id: str = Field(..., description="Stable ID, format: '<source_key>-<hash-or-native-id>'")
 65    title: str
 66    summary: str = Field(default="", description="Short description, max ~400 chars")
 67    url: HttpUrl
 68    source: Source
 69    source_key: str = Field(..., description="Specific fetcher key, e.g. 'anthropic-newsroom'")
 70    category: list[Category] = Field(default_factory=list)
 71    published_at: datetime | None = Field(
 72        default=None, description="Source-published timestamp when known."
 73    )
 74    discovered_at: datetime = Field(
 75        default_factory=lambda: datetime.now(tz=UTC),
 76        description="Timestamp when this server discovered or parsed the item.",
 77    )
 78    sort_at: datetime | None = Field(
 79        default=None,
 80        description="Timestamp used for ordering; normally published_at, otherwise discovered_at.",
 81    )
 82    date_confidence: DateConfidence = DateConfidence.EXACT
 83    importance: Literal[1, 2, 3] = 1
 84    tags: list[str] = Field(default_factory=list)
 85    author: str | None = None
 86    source_type: SourceType = SourceType.OFFICIAL
 87    evidence_tier: EvidenceTier = EvidenceTier.HIGH
 88    is_official: bool = True
 89
 90    @model_validator(mode="after")
 91    def _fill_derived_fields(self) -> "NewsItem":
 92        if self.sort_at is None:
 93            self.sort_at = self.published_at or self.discovered_at
 94        if self.published_at is None and self.date_confidence == DateConfidence.EXACT:
 95            self.date_confidence = DateConfidence.UNKNOWN
 96        self.is_official = self.source_type in {SourceType.OFFICIAL, SourceType.DOCS}
 97        return self
